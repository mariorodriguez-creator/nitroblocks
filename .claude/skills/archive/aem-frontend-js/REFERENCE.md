## Full Component JS Skeleton

```javascript
import { register } from '@netcentric/component-loader';
import config from './dxn-component.config';

class DxnComponent {
  constructor(el, params) {
    // el → DOM element with data-nc="DxnComponent"
    // params → parsed JSON from data-nc-params-DxnComponent
    this.el = el;
    this.config = { ...config, ...params };
    this.selectors = this.config.selectors;
    this.classes = this.config.classes;
    this.init();
  }

  init() {
    this.bindEvents();
    this.setupAccessibility();
  }

  bindEvents() {
    // Event delegation — one listener on root
    this.el.addEventListener('click', this.handleClick.bind(this));
    this.el.addEventListener('keydown', this.handleKeydown.bind(this));
  }

  setupAccessibility() {
    // Set ARIA attributes, live regions
    const announcer = document.createElement('div');
    announcer.setAttribute('aria-live', 'polite');
    announcer.setAttribute('aria-atomic', 'true');
    announcer.className = 'sr-only';
    document.body.appendChild(announcer);
    this.announcer = announcer;
  }

  handleClick(e) {
    if (e.target.matches(this.selectors.trigger)) {
      this.open();
    }
  }

  handleKeydown(event) {
    switch (event.key) {
      case 'Enter':
      case ' ':
        this.activate();
        break;
      case 'Escape':
        this.close();
        break;
      default:
        break;
    }
  }

  open() {
    this.el.classList.add(this.classes.active);
    this.el.setAttribute('aria-expanded', 'true');
  }

  close() {
    this.el.classList.remove(this.classes.active);
    this.el.setAttribute('aria-expanded', 'false');
  }

  activate() { /* ... */ }

  destroy() {
    this.el.removeEventListener('click', this.handleClick);
    this.el.removeEventListener('keydown', this.handleKeydown);
    if (this.announcer) this.announcer.remove();
  }
}

register({ DxnComponent });
```

## Build Commands

Run from `digitalxn-aem-base/digitalxn-aem-base-clientlibs-apps/frontend/`:

```bash
npm run build        # Full build
npm run build:css    # CSS only
npm run lint:js      # ESLint
npm run lint:css     # Stylelint
```
